## Example

export const exampleCode = `
// Create a bullet maker
const makeBullet = extendMaker(makeObject, () => {
  power: 1,
}, (opt) => [
  area({ shape: new Rect(vex2(0), opt.size, opt.size) }),
  damage(opt.power),
]);

// Use maker
add(makeBullet({
  pos: vec2(10, 10),
  power: 4,
}));

// As a children
gun.add(makeBullet({
  pos: vec2(20, 20),
  power: gun.power,
}));
`;
